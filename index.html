from pyscript import Element def process_number(x): if x > 10: result = x * 2 if result > 100: result = result / 3 else: result = x / 2 return result def run_python(event=None): input_value = Element("input-x").element.value if input_value: x = float(input_value) result = process_number(x) Element("output").element.innerText = f"Result: {result}" # Bind the function to the button from js import document document.getElementById("run-button").addEventListener("click", run_python)
